Cliente

CREATE TABLE Cliente(
IDCliente INT NOT NULL IDENTITY,
CPF INT CHAR(11) NOT NULL,
Nome VARCHAR(40) NOT NULL,
IDBC INT NOT NULL DEFAULT 'Ainda não informado',
Telefone INT CHAR(11),
Email VARCHAR(50) NOT NULL,
Senha VARCHAR(20) NOT NULL,
UNIQUE(CPF),
PRIMARY KEY(IDCliente),
FOREIGN KEY(IDBC) REFERENCES DadosBC
)

Vendedor

CREATE TABLE Vendedor(
IDVendedor INT NOT NULL IDENTITY,
CNPJ CHAR(14) NOT NULL,
CPF CHAR(11) NOT NULL,
Nome VARCHAR(40) NOT NULL,
IDBC INT NOT NULL DEFAULT 'Ainda não informado',
Telefone INT CHAR(11),
Email VARCHAR(50) NOT NULL,
SENHA VARCHAR(20) NOT NULL,
PRIMARY KEY (IDVendedor),
UNIQUE(CPF),
FOREIGN KEY(IDBC) REFERENCES DadosBC
)

Dados Bancários

CREATE TABLE DadosBC(
IDBC INT NOT NULL IDENTITY,
IDCliente INT NOT NULL DEFAULT 'Usuário é um vendedor ',
IDVendedor INT NOT NULL DEFAULT 'Usuário é comprador',
NConta CHAR(8) NOT NULL,
NCartao CHAR(16) NOT NULL,
NCCV CHAR(3) NOT NULL,
NBanco VARCHAR(15) NOT NULL,
PRIMARY KEY(IDBC),
FOREIGN KEY(IDCliente) REFERENCES Cliente,
FOREIGN KEY(IDVendedor) REFERENCES Vendedor
)

Produtos 

CREATE TABLE Produto(
IDProduto INT NOT NULL IDENTITY,
IDVendedor INT NOT NULL,
IDCliente INT NOT NULL DEFAULT 'Produto ainda não conprado',
Nome VARCHAR(25) NOT NULL,
Descrição VARCHAR(200),
Preco NUMERIC(4,2) NOT NULL,
Estoque INT NOT NULL,
PRIMARY KEY(IDProduto),
FOREIGN KEY(IDVendedor) REFERENCES Vendedor,
FOREIGN KEY(IDCliente) REFERENCES Cliente
)

Endereço

CREATE TABLE Endereco(
IDEndereco INT NOT NULL IDENTITY,
IDCliente INT NOT NULL IDENTITY,
Rua VARCHAR(45) NOT NULL,
Bairro VARCHAR(45) NOT NULL,
CEP INT CHAR(8) NOT NULL,
NCasa INT VARCHAR(5) NOT NULL,
PRIMARY KEY (IDEndereco),
FOREIGN KEY (IDCliente) REFERENCES Cliente
)
